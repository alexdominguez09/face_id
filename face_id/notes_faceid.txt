Face id
=======

To run app

cd /home/alex/Downloads/face_id/face_id
conda activate faceid

# List all enrolled faces                                                                    
     python -m cli.main list-faces                                                                
     python -m cli.main list-faces --search "John"                                                
                                                                                                  
     # Delete a face                                                                              
     python -m cli.main delete-face --id 1                                                        
     python -m cli.main delete-face --name "John Doe"                                             
                                                                                                  
     # Process a video file                                                                       
     python -m cli.main process-video --input video.mp4 --output result.mp4                       
                                                                                                  
     # Show database statistics                                                                   
     python -m cli.main stats                                                                     
                                                                                                  
     # Export face database                                                                       
     python -m cli.main export-faces --format json --output faces.json                            
                                                                                                  
     # Search faces                                                                               
     python -m cli.main search --name "John"                                                      
     python -m cli.main search --id 1                                                             
                                                                                                  
     # Batch enroll faces from directory                                                          
     python -m cli.main batch-enroll --directory /home/alex/Downloads/people/train/images                               
                                                                                                  
     # Import faces from another database                                                         
     python -m cli.main import-faces --source other.db
